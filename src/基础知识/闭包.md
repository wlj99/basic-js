# 闭包

能够访问另一个函数作用域的变量的函数。即：闭包就是一个函数，这个函数能够访问其他函数的作用域中的变量

```js
function makeFunc() {
  let name = "Mozilla";
  function displayName() {
    console.log(name);
  }
  return displayName;
}

let myFunc = makeFunc();
myFunc();
```

## 使用场景

1. 用闭包模拟私有方法

```js
let makeCounter = (function () {
  let privateCounter = 0;
  function changeBy(val) {
    privateCounter += val;
  }
  return {
    increment: function () {
      changeBy(1);
    },
    decrement: function () {
      changeBy(-1);
    },
    value: function () {
      return privateCounter;
    },
  };
})();
let Counter1 = makeCounter();
let Counter2 = makeCounter();

console.log(Counter1.value()); //  0
Counter1.increment();
Counter1.increment();
console.log(Counter1.value()); // 2
Counter1.decrement();
console.log(Counter1.value()); // 1
console.log(Counte12.value()); // 0
```

> 每个闭包都是引用自己词法作用域内的变量

2. 用闭包解决递归调用问题

```js
function factorial(num) {
  if (num <= 1) {
    return 1
  } else {
    return num * factorial(num - 1)
  }
}
let anotherFactorial = factorial;
factorial = null;
anotherFactorial(4);
/* 报错 factorial is not a function
 可以使用 return num* arguments.callee（num-1），arguments.callee指向当前执行函数，
 在严格模式下不能使用该属性也会报错，所以借助闭包来实现 */

// 使用闭包实现递归
function newFactorial = (function f(num) {
  if (num < 1) {
    return 1
  } else {
    return num * f(num - 1)
  }
}) //实际上起作用的是闭包函数f，而不是外面的函数newFactorial
```

## 使用闭包常见的坑

1. 循环中使用闭包一个常见错误

```js
function setupHelp() {
  let res = [];
  for (var i = 0; i < 3; i++) {
    result[i] = function () {
      console.log(i);
    };
  }
  return res;
}

let res = setupHelp(); // [3, 3, 3]
```

> 使用 var 进行声明，由于变量提升，所以具有函数作用域

解决方案：

```js
function setupHelp() {
  let res = [];
  for (let i = 0; i < 3; i++) {
    res[i] = (function () {
      return i;
    })(i);
  }
  return res;
}

let res = setupHelp(); // [0, 1, 2]
```

> 使用 let 而不是 var，因此每个闭包都绑定了块作用域的变量

2. this 指向

```js
let object = {
  name: "object",
  getName: function () {
    return function () {
      console.info(this.name);
    };
  },
};
object.getName()(); //  undefined  this -> window
```

3. 内存泄漏

```js
function showId() {
  let el = document.getElementById("app");
  el.onclick = function () {
    console.log(el.id); // 这样会导致闭包引用外层的el，当执行完showId后，el无法释放
  };
}

// 改成下面
function showId() {
  let el = document.getElementById("app");
  let id = el.id;
  el.onclick = function () {
    console.log(id);
  };
  el = null; // 主动释放el
}
```
